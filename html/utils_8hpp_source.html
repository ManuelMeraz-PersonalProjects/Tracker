496  typename Storable,
497  typename std::enable_if_t<
498  std::is_base_of_v<database::Storable, std::decay_t<Storable>>, int>>
499 auto database::utils::get_new_id() -> int
500 {
501  auto &storables = utils::retrieve_all<Storable>();
502 
503  int new_id = 0;
504  // Empty or first value is not 1
505  if ((storables.size() >= 1 && storables[0].id() != 1) || storables.empty()) {
506  new_id = 1;
507 
508  // First element is equal to one, can't use adjacent find
509  } else if (storables.size() == 1 && storables[0].id() == 1) {
510  new_id = 2;
511  } else {
512  // Find a gap in the sorted vector of IDs
513  // if a gap exists, this is a deleted key.
514  // storables is sorted
515  auto const is_deleted_key = [](Storable const &s, Storable const &s_next) {
516  return (s_next.id() - s.id()) > 1;
517  };
518 
519  auto potential_id =
520  std::adjacent_find(begin(storables), end(storables), is_deleted_key);
521 
522  if (potential_id != end(storables)) {
523  new_id = (*potential_id).id() + 1;
524  } else {
525  new_id = storables.size() + 1;
526  }
527  }
528 
529  return new_id;
530 }

622  typename Storable,
623  typename std::enable_if_t<
624  std::is_base_of_v<database::Storable, std::decay_t<Storable>>, int>>
625 inline auto database::utils::retrieve_all()
626  -> std::vector<Storable, struct Storable::Allocator> &
627 {
628  static std::vector<Storable, struct Storable::Allocator> storables;
629 
630  if (!data_is_loaded<Storable>) {
631  data_is_loaded<Storable> = true;
632  auto const table_name = utils::type_to_string<Storable>();
633 
634  auto &sql_connection = Database::get_connection();
635  size_t num_rows = utils::count_rows<Storable>();
636  if (num_rows == 0) return storables;
637 
638  std::stringstream sql_command;
639  sql_command << "SELECT * from " << table_name;
640 
641  soci::row from_row;
642  soci::statement statement =
643  (sql_connection.prepare << sql_command.str(), soci::into(from_row));
644  try {
645  statement.execute();
646  } catch (soci::sqlite3_soci_error const &error) {
647  std::cerr << error.what() << std::endl;
648  std::cerr << sql_command.str() << std::endl;
649  throw std::runtime_error(
650  " Failed to retrieve all storables from database");
651  }
652 
653  // Schema contains the column names to map the data correctly
654  std::vector<ColumnProperties> schema;
655  schema.reserve(from_row.size());
656 
657  std::vector<Row> to_rows;
658  to_rows.reserve(num_rows);
659 
660  while (statement.fetch()) {
661  // Getting data from soci row to tracker row
662  Row to_row;
663  to_row.row_data.reserve(from_row.size());
664 
665  for (size_t i = 0; i < from_row.size(); ++i) {
666  soci::column_properties const &props = from_row.get_properties(i);
667 
668  if (schema.size() < from_row.size()) {
669  ColumnProperties column_property;
670  column_property.name = props.get_name();
671  schema.emplace_back(column_property);
672  }
673 
674  switch (props.get_data_type()) {
675  case soci::dt_double:
676  to_row.row_data.emplace_back(from_row.get<double>(i));
677  break;
678  case soci::dt_string:
679  to_row.row_data.emplace_back(from_row.get<std::string>(i));
680  break;
681  case soci::dt_integer:
682  to_row.row_data.emplace_back(from_row.get<int>(i));
683  break;
684  case soci::dt_long_long:
685  to_row.row_data.emplace_back(from_row.get<long long>(i));
686  break;
687  case soci::dt_unsigned_long_long:
688  to_row.row_data.emplace_back(from_row.get<unsigned long long>(i));
689  break;
690  case soci::dt_date:
691  to_row.row_data.emplace_back(from_row.get<std::tm>(i));
692  break;
693  default:
694  throw std::runtime_error("Invalid variant type get!");
695  }
696  }
697 
698  to_rows.emplace_back(to_row);
699  }
700 
701  // This is done after fetching the data from the rows
702  // because while(statement.fetch()) goes into an infinite
703  // loop if this is done inside the loop because we insert/update
704  // storable data as they're constructed
705  storables.reserve(num_rows);
706  for (auto const &row : to_rows) {
707  storables.emplace_back(schema, row);
708  }
709  }
710 
711  return storables;
712 }
